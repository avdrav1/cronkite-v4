## Data Structures

```typescript
interface Feed {
  id: string;
  name: string;
  url: string;
  folderId: string | null;
  folderName: string | null;
  priority: 'high' | 'medium' | 'low';
  status: 'active' | 'paused' | 'error';
  errorMessage?: string;
  lastSyncAt: Date | null;
  articlesPerDay: number;
  customPollingInterval?: number; // minutes, null = use default
}

interface Folder {
  id: string;
  name: string;
  feedCount: number;
  isExpanded: boolean;
}

interface PollingSettings {
  globalInterval: 5 | 15 | 30 | 60 | 'manual';
  adaptiveFrequency: boolean;
  respectTTL: boolean;
  activeHoursOnly: boolean;
  activeHoursStart?: string; // "09:00"
  activeHoursEnd?: string;   // "23:00"
}

interface DigestSettings {
  enabled: boolean;
  deliveryTime: string;      // "07:00"
  timezone: string;          // "America/New_York"
  articleCount: number;
  includeTopArticles: boolean;
  includeAISummary: boolean;
  includeStarred: boolean;
  includeUnreadCount: boolean;
  deliveryMethod: 'email' | 'inapp' | 'both';
  email: string;
}

interface AISettings {
  summariesEnabled: boolean;
  summaryStyle: 'bullets' | 'paragraph' | 'detailed';
  autoSummarize: boolean;
  clusteringEnabled: boolean;
  clusteringSchedule: '6h' | '12h' | 'daily' | 'manual';
  clusteringTime?: string;   // "06:00" if daily
  minArticlesPerCluster: number;
  clusterTimeframe: '12h' | '24h' | '3d' | '7d';
  showClusterBadges: boolean;
  groupByCluster: boolean;
}

interface Cluster {
  id: string;
  title: string;
  articleCount: number;
  sources: string[];
  createdAt: Date;
}
```

- **Tab switch**: Fade content, instant nav highlight
- **Accordion expand**: Smooth height animation (300ms)
- **Toggle switch**: Spring animation on thumb
- **Modals**: Fade in backdrop, scale up content
- **Toast notifications**: Slide in from top-right

- **Desktop**: Sidebar nav + content area
- **Tablet**: Top tabs + content below
- **Mobile**: Full-width, collapsible sections, bottom sheet modals
